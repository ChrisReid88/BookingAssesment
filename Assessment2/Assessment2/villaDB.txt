DROP TABLE IF EXISTS `guest`;
DROP TABLE IF EXISTS `booking`;
DROP TABLE IF EXISTS `customer`;

CREATE TABLE `customer` (
	`customerRef` INT(11) NOT NULL AUTO_INCREMENT,
	`name` VARCHAR(20),
	`address` VARCHAR(100),
	PRIMARY KEY (`customerRef`)
);

CREATE TABLE `booking`(
	`bookingRef` INT(11) NOT NULL AUTO_INCREMENT,
	`arrival` DATE NOT NULL,
	`departure` DATE NOT NULL,
	`customerRef` INT(11),
	PRIMARY KEY(`bookingRef`),
	FOREIGN KEY (`customerRef`) REFERENCES `customer`(`customerRef`)
	);

CREATE TABLE `guest` (
	`passportNo` VARCHAR(10) NOT NULL,
	`guestName` VARCHAR(20),
	`age` INT(3),
	`bookingRef` INT(11),
	PRIMARY KEY(`passportNo`),
	FOREIGN KEY(`bookingRef`) REFERENCES `booking`(`bookingRef`)
);
